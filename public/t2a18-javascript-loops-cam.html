2. `while`

let c=0; while (c < 4) { output(c); c++; }

3. `do...while`

let i=10, r=0; do { output("Ran once: "+r); r++; } while (i < 10);

const p=[2,3,5,7]; for (const n of p) { output(`Prime: ${n}`); }
